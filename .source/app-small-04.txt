async function showThread(id){const c=state.data.conversations.find(x=>x.id===id);$('#thread').innerHTML=`<div class="row"><h2>${esc(c.contactName)}</h2>${c.requiresHandoff?'<span class="pill warn">Handoff required</span>':''}</div><div class="messages">${c.messages.map(m=>`<div class="message ${m.from}">${esc(m.text)}</div>`).join('')}</div><form id="replyForm"><textarea name="text" required placeholder="Send a grounded reply"></textarea><button>Send</button></form>`;$('#replyForm').onsubmit=async e=>{e.preventDefault();await api(`/api/conversations/${id}/messages`,{method:'POST',body:formObj(e.target)});await reload();state.route='inbox';render();showThread(id)}}
document.addEventListener('submit',async e=>{try{if(e.target.id==='loginForm'){e.preventDefault();const x=await api('/api/auth/login',{method:'POST',body:formObj(e.target)});state.csrf=x.csrfToken;return boot()}if(e.target.id==='modalForm'){e.preventDefault();await api('/api/prospects',{method:'POST',body:formObj(e.target)});closeModal();return reload('Prospect added')}if(e.target.id==='clientForm'){e.preventDefault();await api(`/api/clients/${e.target.dataset.id}`,{method:'PATCH',body:formObj(e.target)});closeModal();return reload('Client saved')}if(e.target.id==='agentForm'){e.preventDefault();await api(`/api/agents/${e.target.dataset.id}`,{method:'PATCH',body:formObj(e.target)});closeModal();return reload('Agent saved')}if(e.target.id==='knowledgeForm'){e.preventDefault();await api('/api/knowledge',{method:'POST',body:formObj(e.target)});closeModal();return reload('Knowledge indexed')}if(e.target.id==='testForm'){e.preventDefault();const x=await api(`/api/agents/${e.target.dataset.id}/test`,{method:'POST',body:formObj(e.target)});$('#testResult').textContent=JSON.stringify(x.result,null,2)}}catch(err){toast(err.message,true)}});
$('#closeModal').onclick=closeModal;$('#logout').onclick=async()=>{await api('/api/auth/logout',{method:'POST',body:{}});location.reload()};boot();
