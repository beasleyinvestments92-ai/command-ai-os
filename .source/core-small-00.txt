function parse(schema, value) {
  const result = schema.safeParse(value);
  if (!result.success) throw new HttpError(400, 'Request validation failed.', result.error.flatten());
  return result.data;
}

async function tenant(req) {
  const row = await one('SELECT * FROM tenants WHERE id=$1', [req.auth.tenant.id]);
  if (!row) throw new HttpError(404, 'Workspace not found.');
  return row;
}

async function bootstrapPayload(req) {
  const tenantRow = await tenant(req);
  const [prospectRows, clientRows, agentRows, conversationRows, documents, integrations] = await Promise.all([
    many('SELECT * FROM prospects WHERE tenant_id=$1 ORDER BY created_at DESC', [tenantRow.id]),
    many('SELECT * FROM clients WHERE tenant_id=$1 ORDER BY created_at DESC', [tenantRow.id]),
    many('SELECT * FROM agents WHERE tenant_id=$1 ORDER BY created_at DESC', [tenantRow.id]),
    many('SELECT * FROM conversations WHERE tenant_id=$1 ORDER BY updated_at DESC LIMIT 100', [tenantRow.id]),
    many(`SELECT kd.id,kd.client_id,kd.agent_id,kd.title,kd.source_type,kd.source_url,kd.status,kd.created_at,kd.updated_at,count(kc.id)::int AS chunks
          FROM knowledge_documents kd LEFT JOIN knowledge_chunks kc ON kc.document_id=kd.id WHERE kd.tenant_id=$1
          GROUP BY kd.id ORDER BY kd.updated_at DESC`, [tenantRow.id]),
    many('SELECT provider,status,metadata,updated_at FROM integrations WHERE tenant_id=$1', [tenantRow.id])
  ]);
  const conversations = [];
  for (const row of conversationRows) conversations.push(await serialize.conversation(row));
  return {
    user: req.auth.user,
    tenant: req.auth.tenant,
    csrfToken: req.auth.csrfToken,
    dashboard: await serialize.dashboard(tenantRow.id, tenantRow),
    brand: tenantRow.brand,
    prospects: prospectRows.map(serialize.prospect),
    clients: clientRows.map(serialize.client),
    agents: agentRows.map(serialize.agent),
    conversations,
    knowledgeDocuments: documents.map((row) => ({ id: row.id, clientId: row.client_id, agentId: row.agent_id, title: row.title, sourceType: row.source_type, sourceUrl: row.source_url, status: row.status, chunks: row.chunks, createdAt: row.created_at, updatedAt: row.updated_at })),
    integrations,
    settings: tenantRow.settings
  };
}

coreRouter.get('/bootstrap', asyncRoute(async (req, res) => res.json(await bootstrapPayload(req))));
coreRouter.get('/dashboard', asyncRoute(async (req, res) => { const row = await tenant(req); res.json(await serialize.dashboard(row.id, row)); }));
