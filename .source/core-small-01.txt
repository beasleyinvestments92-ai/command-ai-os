    { key: 'authentication', label: 'Secure authentication and tenant isolation', ready: true },

    { key: 'agents', label: 'At least one active agent', ready: Number(agents.active) > 0 },
    { key: 'clients', label: 'At least one live client', ready: Number(clients.live) > 0 },
    { key: 'twilio', label: 'Voice and SMS provider', ready: Boolean(config.twilio.accountSid && config.twilio.authToken) },

    { key: 'encryption', label: 'Integration secret encryption', ready: Boolean(config.integrationEncryptionKey) }

coreRouter.post('/prospects', requireRole('owner','admin','member'), asyncRoute(async (req, res) => {
  const body = parse(prospectSchema, req.body);
  const row = await one(`INSERT INTO prospects(id,tenant_id,business_name,industry,city,website,phone,email)
    VALUES($1,$2,$3,$4,$5,$6,$7,$8) RETURNING *`, [id('pros'), req.auth.tenant.id, cleanText(body.businessName,120), cleanText(body.industry,80), cleanText(body.city,100), cleanText(body.website,500), normalizePhone(body.phone), normalizeEmail(body.email)]);
  await auditLog(req, 'prospect.created', 'prospect', row.id, { businessName: row.business_name });
  res.status(201).json({ prospect: serialize.prospect(row) });
}));
